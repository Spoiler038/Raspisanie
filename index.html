// Отправка уведомления в Telegram - через Edge Function async function sendTelegramNotification(lesson, interval) { if (!telegramConnected || !telegramChatId || !currentUser) return; try { // Получаем свежую сессию для токена const { data: { session } } = await supabaseClient.auth.getSession(); if (!session) { console.error('Нет активной сессии'); return; } const timeStr = lesson.time.substring(0, 5); const durationHours = Math.floor(lesson.duration / 60); const durationMins = lesson.duration % 60; const durationStr = durationHours > 0 ? `${durationHours}ч ${durationMins > 0 ? durationMins + 'м' : ''}` : `${durationMins}м`; const intervalText = { 'За 30 минут': '30 минут', 'За 1 час': 'часа', 'За 6 часов': '6 часов', 'За 12 часов': '12 часов', 'За день': '24 часа' }[interval] || interval; const message = `📅 Напоминание о занятии\n\n` + `⏰ До начала: ${intervalText}\n` + `👤 Ученик: ${lesson.student}\n` + `🕐 Время: ${timeStr}\n` + `⏱️ Длительность: ${durationStr}\n` + (lesson.notes ? `📝 Заметки: ${lesson.notes}` : ''); // Отправляем через Edge Function const response = await fetch( `${SUPABASE_URL}/functions/v1/telegram-send`, { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${session.access_token}` }, body: JSON.stringify({ chatId: telegramChatId, message: message, lessonId: lesson.id, interval: interval }) } ); const result = await response.json(); if (result.success) { console.log('✅ Telegram notification sent via Edge Function'); } else { console.error('❌ Error from Edge Function:', result.error); } } catch (error) { console.error('❌ Ошибка отправки Telegram уведомления:', error); } }
